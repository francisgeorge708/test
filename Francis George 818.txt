<script>
    function calculate(){    
    var investment = ParseFloat(document.getElementById("investment").value);
    var interest =  ParseFloat(document.getElementById("interest").value);
    var years = ParseFloat(document.getElementById("years").value);

    if(isNaN(investment) || isNaN(interest) || isNaN(years) || (investment<=0) || (interest<=0) || (years<=5))
    {
      if(isNaN(investment) || (investment<=0))
        alert("Please enter numeric value for investment and should be greater than 0");
      if(isNaN(interest) || (interest<=0))
      alert("Please enter numeric value for interest and should be greater than 0");
      if(isNaN(years) || (years<=5))
      alert("Please enter numeric value for years and should be greater than 5");      
    }
    else
    {
        document.getElementById("future_value").innerHTML = (investment*(1 + (investment*(interest))*years)).toFixed(2);
    }
  }
</script>
